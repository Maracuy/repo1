CREATE TABLE IF NOT EXISTS PETICIONES_PROGRAMAS(
	ID INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    NOMBRE_COMPLETO VARCHAR(100) NOT NULL,
    PETICION VARCHAR(150) NOT NULL,
    FOLIO VARCHAR(150) NOT NULL,
    PRIORIDAD VARCHAR(150) NOT NULL,
    DESTINO VARCHAR(150) NOT NULL,
    ESTADO VARCHAR(150) NOT NULL,
    FECHA_INICIO VARCHAR(150) NOT NULL,
    FECHA_FINAL VARCHAR(150) NOT NULL
)ENGINE=INNODB;
